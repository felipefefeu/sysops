AWSTemplateFormatVersion: 2010-09-09

Description: Template to create an EC2 instance and enable SSH

Parameters: 
  ImageID:
    Description: Id of image
    Type: AWS::EC2::Image::Id
    Default: ami-0fa399d9c130ec923
    ConstraintDescription: Provide the name of an existing AMI

  InstanceTypeParameter:
    Type: String
    Default: t3.micro
    AllowedValues:
      - t2.micro
      - t3.micro
      - m1.large
    Description: Enter t3.micro, t2.micro, or m1.large. Default is t3.micro.

Resources:

  MyEC2Instance:
    Type: 'AWS::EC2::Instance'
    Properties:
      InstanceType: !Ref InstanceTypeParameter
      ImageId: !Ref ImageID
      SecurityGroups:
        - !Ref InstanceSecurityGroup
        - !Ref InstanceSecurityGroup2

      Tags:
        - Key: Name
          Value: takmandu

  InstanceSecurityGroup:
    Type: 'AWS::EC2::SecurityGroup'
    Properties:
      GroupName: MyDMZSecurityGroup
      GroupDescription: Enable SSH access via port 22
      SecurityGroupIngress:
        IpProtocol: tcp
        FromPort: 22
        ToPort: 22
        CidrIp: 0.0.0.0/0
 
  InstanceSecurityGroup2:
    Type: 'AWS::EC2::SecurityGroup'
    Properties:
      GroupName: http
      GroupDescription: Enable http access via port 80
      SecurityGroupIngress:
        IpProtocol: tcp
        FromPort: 80
        ToPort: 80
        CidrIp: 0.0.0.0/0

Outputs: 
  InstanceID:
    Description: The Instance ID
    Value: !Ref MyEC2Instance
